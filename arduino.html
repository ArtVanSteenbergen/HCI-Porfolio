BLINK MULTIPLE LEDS

int greenLedPin = 10;
int yellowLedPin = 9;
void setup() {
	pinMode(greenLedPin, OUTPUT);
	inMode(yellowLedPin, OUTPUT);
}
void loop() {
	analogWrite(greenLedPin, 0);
	analogWrite(yellowLedPin, 0);
	delay(500);
	analogWrite(yellowLedPin, 255);
	delay(500);
	analogWrite(greenLedPin, 255);
	analogWrite(yellowLedPin, 0);
	delay(500);
	analogWrite(yellowLedPin, 255);
	delay(500);
}

FADE LED - ANALOG OUTPUT

int greenLedPin = 10;
int yellowLedPin = 9;
void setup() {
	pinMode(greenLedPin, OUTPUT);
	pinMode(yellowLedPin, OUTPUT);
}
void loop() {
	 for (int brightness=0; brightness <256; brightness++){
		analogWrite(greenLedPin, brightness);
		analogWrite(yellowLedPin, 255 - brightness);
		delay(10);
	}
}

CONTROLLING A LED WITH A POTMETER

float sensorValue = 0;
int sensorPin = A0;
int greenLedPin = 10;
void setup() {
	Serial.begin(9600);
	pinMode(sensorPin, INPUT);
	pinMode(greenLedPin, OUTPUT);
}
void loop() {
	sensorValue = analogRead(sensorPin);
	sensorValue = (sensorValue/1023)*255;
	analogWrite(greenLedPin, sensorValue);
	Serial.println(sensorValue);
	delay(100);
}

This excercise used the same code as the previous one. The only difference to the breadboard is that a resistor was replaced with another LDR. In this assignment we needed to explain what happens when the resisor is replaced.

We can calculate what happens with this formula:

Vout = Vin * R2:(R1 * R2) = Vout

For example we can fill in this formula with differend light inputs to calculate the voltage output.

Left resistor in the dark and right resistor in the light:

R1 = 5megohm
R2 = 200ohm
Vin = 5v
5v * 200ohm : (5megohm + 200ohm) = ~0v

Left resistor in the light and right resistor in the dark:

R1 = 200ohm
R2 = 5megohm
Vin = 5v
5v * 5megohm : (200ohm + 5megohm) = ~5v

Both resistors in the light:

R1 = 200ohm
R2 = 200ohm
Vin = 5v
5v * 200ohm : (200ohm + 200ohm) = 2.5v

Both resistors in the dark:

R1 = 5megohm
R2 = 5megohm
Vin = 5v
5v * 5megohm : (5megohm + 5megohm) = 2.5v

import processing.serial.*;
Serial myPort;
String sensorReading="";
void setup() {
	size(400, 400);
	myPort = new Serial(this, Serial.list()[2], 9600);
	myPort.bufferUntil('\n');
}
void draw() {
	background(255);
	fill(float(sensorReading),0,255-float(sensorReading));
	noStroke();
	text("Sensor Reading: " + sensorReading, 20, 20);
	rect(map(float(sensorReading),0,255,0,200),map(sin(float(sensorReading)/10),0,1,100,120), 200, 200);
}
void serialEvent (Serial myPort) {
	sensorReading = myPort.readStringUntil('\n');
}

BUTTONS

int ledPin = 13;
int buttonPinL = 3;
int buttonPinR = 2;
int readingL;
int readingR;
void setup() {
	pinMode(ledPin, OUTPUT);
	pinMode(buttonPinL, INPUT);
	pinMode(buttonPinR, INPUT);
}
void loop(){
	readingL = digitalRead(buttonPinL);
	readingR = digitalRead(buttonPinR);
	if ((readingL == HIGH && readingR == LOW) || (readingL == LOW && readingR == HIGH)) {
		digitalWrite(ledPin, HIGH);  
	} else {
		digitalWrite(ledPin, LOW);
	}
}

CONTROLLING THE SERVO WITH INPUTS

#include 
Servo myServo;
int servoPin = 9;
int pos = 0;
int buttonPinL = 3;
int buttonPinR = 2;
void setup() {
	myServo.attach(servoPin);
	pinMode(buttonPinL, INPUT);
	pinMode(buttonPinR, INPUT);
}
void loop() {
		if (digitalRead(buttonPinL) == HIGH && digitalRead(buttonPinR) == LOW){
			pos++;
		  	if (pos > 160) {
		    	pos = 160;
		  	}
		 	delay(15); 
		}
		if (digitalRead(buttonPinR) == HIGH && digitalRead(buttonPinL) == LOW){
			pos--;
			if (pos < 0) {
				pos = 0;
			}
			delay(15);
		}
		if (pos >= 0 && pos <= 160) {
			myServo.write(pos);
		}
}
